int main()
{
    
    int a[20], n, i;    
 
n=1248;    
int token=0;
int gap=0;
int maxGap=0;
for(i=0; n>0; i++)    
{    
a[i]=n%2;  
cout << n%2;
if (n%2 == 1 && token==0)
  token = 1;
else if (n%2 == 0 && token==1)
     gap++;
  else if (token==1 && gap>0) 
    {token=0;
    if (gap > maxGap )
       {maxGap= gap;
       gap=0;}
       else
       gap=0;
    } 
n= n/2;  
}    
cout <<endl;
for(i=i-1 ;i>=0 ;i--)    
{    
cout<<a[i];    
}    

cout << endl;

}
